## EXECUTIVE OVERVIEW

### Project Scope
RitualFeed is transitioning from an MVP to a **production-grade autonomous research digest platform** on Ritual Chain. This playbook incorporates 20 stakeholder enhancement requests across three strategic phases:

- **Phase 7: Observability** - Logging, metrics, monitoring, alerting
- **Phase 8: AI Intelligence** - Embeddings, vector search, recommendations, RAG
- **Phase 9: Production Scale** - Caching, indexers, load testing, scaling

### What Gets Built vs What Stays Blockchain-Side

| Component | Lovable Builds | Ritual/Backend Owns |
|-----------|-----------------|-------------------|
| UI/UX | ✅ All screens, modals, forms | - |
| Real-time Feed | ✅ Display + Live Updates | ✅ DigestAgent Contract |
| Search Engine | ✅ Frontend filters | ✅ Indexer (Graph/Subsquid) |
| Recommendations | ✅ Render + Preferences UI | ✅ Vector DB + ML Pipeline |
| RAG Chatbot | ✅ UI + Message Threading | ✅ LLM + Retrieval Logic |
| Notifications | ✅ Toast + UI Modals | ✅ Backend (Email/Telegram) |
| Wallet Integration | ✅ Connect + Transactions | ✅ Contract Calls |
| Analytics Dashboard | ✅ Charts + Tables | ✅ Event Indexing |
| Caching | ✅ React Query/SWR | ✅ Redis (Backend) |
| Revenue Tiers | ✅ UI + Paywalls | ✅ Smart Contract Gates |

import axios, { AxiosInstance, AxiosError } from 'axios'

const API_BASE_URL = process.env.NEXT_PUBLIC_BACKEND_API || 'http://localhost:3001'

class ApiClient {
  private client: AxiosInstance

  constructor() {
    this.client = axios.create({
      baseURL: API_BASE_URL,
      timeout: 30000,
      headers: { 'Content-Type': 'application/json' },
    })

    // Add interceptor for error handling
    this.client.interceptors.response.use(
      (response) => response.data,
      (error: AxiosError) => {
        console.error('API Error:', error.response?.status, error.message)
        return Promise.reject({
          status: error.response?.status,
          message: error.response?.data?.message || 'API request failed',
        })
      },
    )
  }

  // Search endpoints
  async search(query: string, filters?: SearchFilters): Promise<SearchResult[]> {
    return this.client.get('/search', { params: { q: query, ...filters } })
  }

  // Recommendations endpoints
  async getRecommendations(userId: string): Promise<Paper[]> {
    return this.client.get(`/recommendations/${userId}`)
  }

  // Chat/RAG endpoints
  async chatWithArchive(message: string, context?: string): Promise<ChatResponse> {
    return this.client.post('/chat', { message, context })
  }

  // Analytics endpoints
  async getUserMetrics(userId: string): Promise<UserMetrics> {
    return this.client.get(`/analytics/user/${userId}`)
  }

  // Citation intelligence endpoints
  async getCitationTrends(): Promise<CitationTrend[]> {
    return this.client.get('/citations/trends')
  }

  // Subscription endpoints
  async updateSubscriptionTier(userId: string, tier: 'free' | 'pro' | 'researcher'): Promise<void> {
    return this.client.post(`/subscriptions/${userId}/tier`, { tier })
  }

  // Notification endpoints
  async updateNotificationPreferences(userId: string, prefs: NotificationPrefs): Promise<void> {
    return this.client.post(`/notifications/${userId}/preferences`, prefs)
  }
}

export const apiClient = new ApiClient()

import React, { useState } from 'react'

interface DigestFeedProps {

export default function DigestFeed({
  isLoading = false,
  hasError = false,
  onRefresh,
}: DigestFeedProps) {
  const [selectedTopic, setSelectedTopic] = useState<string | null>(null)
  const [sortBy, setSortBy] = useState<'recent' | 'score'>('recent')

  // Mock papers with filtering
  const filteredPapers = selectedTopic
    ? MOCK_PAPERS.filter((p) => p.topics.includes(selectedTopic))
    : MOCK_PAPERS

  const sortedPapers = [...filteredPapers].sort((a, b) => {
    if (sortBy === 'score') {
      const scoreMap = { high: 3, medium: 2, low: 1 }
      return scoreMap[b.score] - scoreMap[a.score]
    }
    return b.publishedAt - a.publishedAt
  })

  // Extract unique topics from mock data
  const allTopics = Array.from(new Set(MOCK_PAPERS.flatMap((p) => p.topics)))

  if (hasError) {

          icon="🔄"
        />
      </div>
    )
  }

import React, { ReactNode } from 'react'

interface ErrorBoundaryProps {
  children: ReactNode
  fallback?: (error: Error, retry: () => void) => ReactNode
}

interface ErrorBoundaryState {
  hasError: boolean
  error: Error | null
}

export default class ErrorBoundary extends React.Component<ErrorBoundaryProps, ErrorBoundaryState> {
  constructor(props: ErrorBoundaryProps) {
    super(props)
    this.state = { hasError: false, error: null }
  }

  static getDerivedStateFromError(error: Error): ErrorBoundaryState {
    return { hasError: true, error }
  }

  componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
    console.error('Error caught:', error, errorInfo)
    // Send to error tracking service (Sentry, etc.)
  }

  handleRetry = () => {
    this.setState({ hasError: false, error: null })
  }

  render() {
    if (this.state.hasError) {
      return (
        this.props.fallback ? (
          this.props.fallback(this.state.error!, this.handleRetry)
        ) : (
          <div className="flex items-center justify-center min-h-screen">
            <div className="text-center">
              <h1 className="text-3xl font-bold text-slate-900 dark:text-white mb-2">
                Something went wrong
              </h1>
              <p className="text-slate-600 dark:text-slate-400 mb-4">
                {this.state.error?.message}
              </p>
              <button
                onClick={this.handleRetry}
                className="px-6 py-2 rounded-lg bg-blue-500 hover:bg-blue-600 text-white font-medium"
              >
                Try Again
              </button>
            </div>
          </div>
        )
      )
    }

    return this.props.children
  }
}

  if (error instanceof ApiError) {

  }
  if (error instanceof Error) {

  }
  return 'An unknown error occurred'

// Prevent XSS attacks
export function sanitizeInput(input: string): string {
  return input
    .replace(/[<>\"']/g, (char) => {
      const escapeMap: { [key: string]: string } = {
        '<': '&lt;',
        '>': '&gt;',
        '"': '&quot;',
        "'": '&#x27;',
      }
      return escapeMap[char] || char
    })
    .trim()
}

// Validate and sanitize search queries
export function sanitizeSearch(query: string): string {
  const sanitized = sanitizeInput(query)
  return sanitized.substring(0, 500) // Max length
}

// URL safety check
export function isSafeUrl(url: string): boolean {
  try {
    const parsed = new URL(url)
    return ['https:', 'http:'].includes(parsed.protocol)
  } catch {
    return false
  }
}

export function trackEvent(eventName: string, properties?: Record<string, any>) {
  if (typeof window === 'undefined') return
  
  // Send to analytics endpoint
  fetch('/api/events', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      eventName,
      properties,
      timestamp: new Date().toISOString(),
    }),
  }).catch(console.error)
}

export function trackPageView(pageName: string) {
  trackEvent('page_view', { page: pageName })
